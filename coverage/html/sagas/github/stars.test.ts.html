'use strict';
 
var _chai = require('chai');
 
var _effects = require('redux-saga/effects');
 
var _TestHelper = require('../../../app/helpers/TestHelper.tsx');
 
var _stars = require('../../../app/sagas/github/stars');
 
var _stars2 = require('../../../app/actions/github/stars');
 
describe('Sagas', function () {
    describe('Github', function () {
        describe('Get Stars', function () {
            var generator = void 0;
            var githubUrl = 'https://api.github.com/repos/barbar/vortigern';
            beforeEach(function () {
                generator = (0, _stars.getGithubStars)({ repo: 'barbar/vortigern' });
            });
            /** 200 */
            it('dispatches Request and Success Actions on OK requests', function (done) {
                // Check if call is made.
                (0, _chai.expect)(generator.next().value).to.deep.equal((0, _effects.call)(_TestHelper.handleRequest, githubUrl));
                // Check if result is ok.
                (0, _chai.expect)(generator.next().value).to.deep.equal((0, _effects.put)((0, _stars2.githubStarsSuccess)(0)));
                done();
            });
            /** 400 */
            it('dispatches Failure on failed requests', function (done) {
                // Check if call is made.
                (0, _chai.expect)(generator.next().value).to.deep.equal((0, _effects.call)(_TestHelper.handleRequest, githubUrl));
                // Check if result is wrong.
                (0, _chai.expect)(generator.throw({}).value).to.deep.equal((0, _effects.put)((0, _stars2.githubStarsFailure)({})));
                done();
            });
        });
    });
});
